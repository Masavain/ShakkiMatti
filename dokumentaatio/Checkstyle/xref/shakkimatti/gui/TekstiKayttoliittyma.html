20      public void kaynnista() {
21          
22          System.out.println(pelilauta.toString());
23  
24          Pelaaja vuoro = Pelaaja.VALKOINEN;
25  
26          while (true) {
27              // valkoisen siirto
28              if (vuoro == Pelaaja.VALKOINEN) {
29                  System.out.println("valkoinen, anna lähtökoordinaatti:");
30              } else if (vuoro == Pelaaja.MUSTA) {
31                  System.out.println("musta, anna lähtökoordinaatti:");
32              }
33              String lahto = lukija.nextLine();
34              String[] koord = lahto.split("");
35  
36              int xMista = kirjaimet.indexOf(koord[0]);
37              int yMista = Integer.parseInt(koord[1]) - 1;
38  
39              if (pelilauta.validiSiirrettava(vuoro, xMista, yMista)) {
40                  vuoro(vuoro, xMista, yMista);
41              }
42              System.out.println(pelilauta.toString());
43  
44              if (vuoro == Pelaaja.MUSTA) {
45                  vuoro = Pelaaja.VALKOINEN;
46              } else {
47                  vuoro = Pelaaja.MUSTA;
48              }
49          }
50      }
51  
52      public void vuoro(Pelaaja pelaaja, int xMista, int yMista) {
53  
54          while (true) {
55              System.out.println("anna päätekoordinaatti:");
56              String paate = lukija.nextLine();
57              String[] koord = paate.split("");
58              int xMihin = kirjaimet.indexOf(koord[0]);
59              int yMihin = Integer.parseInt(koord[1]) - 1;
60              if (pelilauta.siirto(xMista, yMista, xMihin, yMihin)) {
61                  break;
62              } else {
63                  System.out.println("ei onnaa.");
64              }
65          }
