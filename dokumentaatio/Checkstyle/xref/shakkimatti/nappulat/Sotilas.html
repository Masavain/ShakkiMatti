1   package shakkimatti.nappulat;
2   
3   import java.util.*;
4   import shakkimatti.logiikka.Pelaaja;
5   
6   /**
7    * Kuvaa shakkilaudan sotilas-nappulaa, jatkaa abstraktia Nappula-luokkaa.
8    *
9    */
10  public class Sotilas extends Nappula {
11  
12      public boolean liikutettu;
13  
14      /**
15       * konstruktori, luo uuden kuninkaan.
16       * @param x x-koordinaatti
17       * @param y y-koordinaatti
18       * @param color pelaaja, jolle nappula kuuluu
19       * @param liikutettu kuvastaa sitä, onko sotilasta liikutettu vai ei, aluksi
20       * yleensä false;
21       */
22      public Sotilas(int x, int y, Pelaaja color, boolean liikutettu) {
23          super(x, y, color);
24          setMerkki("S");
25          this.liikutettu = liikutettu;
26      }
27  
28      @Override
29      public List<String> mahdollisetSiirrot(Nappula[][] tilanne) {
30          List<String> siirrot = new ArrayList<String>();
31          int x = getX();
32          int y = getY();
33          if (this.getPelaaja() == Pelaaja.MUSTA) {
34              if (y < 7) {
35                  if (tilanne[x][y + 1] == null) {
36                      siirrot.add(x + "," + (y + 1));
37                  }
38                  if (y < 6 && !this.liikutettu && tilanne[x][y + 2] == null) {
39                      siirrot.add(x + "," + (y + 2));
40                  }
41              }
42          } else if (this.getPelaaja() == Pelaaja.VALKOINEN) {
43              if (y > 0) {
44                  if (tilanne[x][y - 1] == null) {
45                      siirrot.add(x + "," + (y - 1));
46                  }
47                  if (y > 1 && !this.liikutettu && tilanne[x][y - 2] == null) {
48                      siirrot.add(x + "," + (y - 2));
49                  }
50              }
51          }
52  
53          if (getPelaaja() == Pelaaja.MUSTA) {
54              if (x < 7 && y < 7) {
55                  if (checkSyotava(x + 1, y + 1, tilanne)) {
56                      siirrot.add((x + 1) + "," + (y + 1));
57                  }
58              }
59              if (x > 0 && y < 7) {
60                  if (checkSyotava(x - 1, y + 1, tilanne)) {
61                      siirrot.add((x - 1) + "," + (y + 1));
62                  }
63              }
64          }
65          if (this.getPelaaja() == Pelaaja.VALKOINEN) {
66              if (x < 7 && y > 0) {
67                  if (checkSyotava(x + 1, y - 1, tilanne)) {
68                      siirrot.add((x + 1) + "," + (y - 1));
69                  }
70              }
71              if (x > 0 && y > 0) {
72                  if (checkSyotava(x - 1, y - 1, tilanne)) {
73                      siirrot.add((x - 1) + "," + (y - 1));
74                  }
75              }
76          }
77          return siirrot;
78      }
79  
80      /**
81       * muuten kuten Nappulan liiku, mutta liikkumisen jälkeen asetetaan sotilaan
82       * liikutettu-muuttuja true:ksi.
83       *
84       */
85      @Override
86      public void liiku(int xMihin, int yMihin, Nappula[][] lauta) {
87          List<String> mahdSiirrot = this.mahdollisetSiirrot(lauta);
88          if (mahdSiirrot.contains(xMihin + "," + yMihin)) {
89              setX(xMihin);
90              setY(yMihin);
91              liikutettu = true;
92          }
93      }
94  
95  }
