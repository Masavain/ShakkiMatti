7       boolean liikutettu;

9       public Sotilas(int x, int y, int color) {
10          super(x, y, color);
11          this.merkki = "S";
12          this.liikutettu = false;
13      }
14  
15      @Override
16      public List<String> mahdollisetSiirrot() {
17          List<String> siirrot = new ArrayList<String>();
18          if (this.color == 0) {
19              if (y < 8) {
20                  siirrot.add(x + "," + (y + 1));
21                  if (!this.liikutettu) {
22                      siirrot.add(x + "," + (y + 2));
23                  }
24              }
25          } else if (this.color == 1) {
26              if (y >= 0) {
27                  siirrot.add(x + "," + (y - 1));
28                  if (!this.liikutettu) {
29                      siirrot.add(x + "," + (y - 2));
30                  }
31              }
32          }
33          return siirrot;
34      }
35  
36      @Override
37      public void liiku(int x, int y) {
38          List<String> mahdSiirrot = this.mahdollisetSiirrot();
39          if (mahdSiirrot.contains(x + "," + y)) {
40              if (!this.liikutettu) {
41                  this.liikutettu = true;
42              }
43              this.y = y;
44          }
45  
46      }
