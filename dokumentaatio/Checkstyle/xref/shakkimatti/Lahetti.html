18                  siirrot.add((x + i) + "," + (y + i));

20              if (x + i < 8 && y - i >= 0) {
21                  siirrot.add((x + i) + "," + (y - i));
22              }
23              if (x - i >= 0 && y + i < 7) {
24                  siirrot.add((x - i) + "," + (y + i));
25              }
26              if (x - i >= 0 && y - i >= 0) {
27                  siirrot.add((x - i) + "," + (y - i));

29          }
